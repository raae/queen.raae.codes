React Query pagination with Gatsby (File System Route API)

client side react query gatsby

For Prune your follows, we already had an URL structure for filters like this:

  • /app/filter/unpopular
  • /app/filter/unfollowed
  • /app/filter/unpopular

But for most filters, there are more than 60 accounts, so we added pagination by:

import React from "react";
import axios from "axios";
import { useQuery } from "@tanstack/react-query";

export default function SearchPage(props) {
  const pageIndex = parseInt(props.page) || 0;
  const size = 60;
  const offset = pageIndex * size; //ğŸ‘ˆğŸ‘ˆğŸ‘ˆ 3ï¸âƒ£
  const { data } = useQuery({
    queryKey: ["accounts", props.filter, props.page], // ğŸ‘ˆğŸ‘ˆğŸ‘ˆ 2ï¸âƒ£
    queryFn: ({ signal }) =>
      axios.get("/api/accounts", {
        params: {
          size: size,
          filter: props.filter,
          offset: offset, //ğŸ‘ˆğŸ‘ˆğŸ‘ˆ 4ï¸âƒ£
        },
        signal: signal,
      }),
    select: (result) => result.data,
  });

  return (
    <ul>
      {(data?.records || []).map((record) => (
        <li key={record.accountId}>
          {record.name} @{record.username}
        </li>
      ))}
    </ul>
  );
}
